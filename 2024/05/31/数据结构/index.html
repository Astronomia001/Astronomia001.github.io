1.冒泡排序
排序次数：至多n-1次，至少1次
算法实现： 

#include <iostream>

#define N0 1001

typedef int ElemType; //元素的类型定义

ElemType array[N0]; //待排序的数据(下标0用作'哨兵'?)

/*(1)排序算法: [冒泡] */
void bubble_sort(ElemType a[], int n){
    int flag; //flag-一趟冒泡排序中是否有发生过元素交换(默认没有0)
    
    ElemType t; //t-临时变量(用于交换) 
    
    /*(A)按升序(从小到大)排序a[]中的元素, 采用的'冒泡'排序算法, 实现(一): 【较大元素从左往右移】*/
    for(int i=0; i < n-1; i++){ //n-1趟排序
        //(A.1)一趟排序开始之前，将flag标识置为0，意即此趟排序过程中没有元素发生过交换
        flag = 0; 
        
        //(A.2)进行一趟"冒泡排序", 遍历j=0 ~ n-1-i的所有元素, 若a[j]比 后一个元素a[j+1]大, 则交换2个元素. 一趟排序的效果是将(0 ~ n-1-i)的最大元素交换到a[n-1-i]位置上!
        for(int j=0; j<n-1-i; j++){
            if( a[j] > a[j+1] ){ //若j位置上的元素 比 j+1位置上的元素大, 则交换a[j]和a[j+1]两个元素
                flag = 1; //此趟排序过程中，有元素发生过交换
                
                t = a[j];
                a[j] = a[j+1];
                a[j+1] = t;
            }
        }

        //(A.3)根据一趟排序过程中是否有元素发生过交换，决定是否进行下一趟排序
        if( !flag ){//此趟排序过程中没有发生任何元素交换，元素已经有序，不需要再排序，提前退出排序过程！
            break;
        }
    }

    /*(B)按升序(从小到大)排序a[]中的元素, 采用的'冒泡'排序算法, 实现(二): 【较小元素从右往左移】 
    for(int i=0; i< n-1; i++){ //n-1趟排序
        //(B.1)一趟排序开始之前，将flag标识置为0，意即此趟排序过程中没有元素发生过交换
        flag = 0; 
        
        //(B.2)进行一趟"冒泡排序", 遍历j=n-1 ~ 1的所有元素, 若a[j]比 前一个元素a[j-1]小, 则交换2个元素. 一趟排序的效果是将(n-1 ~ i+1)的最小元素交换到a[i]位置上!
        for(int j=n-1; j>i; j--){
            if( a[j] < a[j-1] ){ //若j位置上的元素 比 j-1位置上的元素小, 则交换两个元素
                flag = 1; //此趟排序过程中，有元素发生过交换
                
                t = a[j];
                a[j] = a[j-1];
                a[j-1] = t;
            }
        }
        
        //(B.3)根据一趟排序过程中是否有元素发生过交换，决定是否进行下一趟排序
        if( !flag ){//此趟排序过程中没有发生任何元素交换，元素已经有序，不需要再排序，提前退出排序过程！
            break;
        }
    }
    */
} 

/*(O)输出结果数组 */
void showInfo(ElemType a[], int n){
    for(int i=0; i<n; i++){
        printf(" %d", a[i]);
    }
    printf("\n");
}
